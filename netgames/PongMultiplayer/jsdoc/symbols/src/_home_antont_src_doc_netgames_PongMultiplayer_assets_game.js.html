 44     /** @param {number} clientId */    
 45     getClientName: function(clientId)
 46     {
 47         var clientIdInt = parseInt(clientId);
 48         if (clientIdInt == 0)
 49             return "CPU";
 50         var thisClient = server.GetUserConnection(clientIdInt);
 51         if (thisClient == null)
 52             return "Unknown";
 53         return thisClient.GetProperty("username");

 56     /** @param {null} */
 57     createGame: function()
 58     { 
 59         data.p1_bat = scene.GetEntityByName("bat_a");
 60         data.p2_bat = scene.GetEntityByName("bat_b");
 61         data.ball = scene.GetEntityByName("ball");
 62         data.serverId = client.GetConnectionID();
 63         
 64         data.running = false;
 65         data.speed = 16;
 66         data.winscore = 2;
 67         data.gameTimeIter = 0.0;
 68         data.p1_client_id = null;
 69         data.p2_client_id = null;
 70         data.p1_score = 0;
 71         data.p2_score = 0;
 72 
 73         gameServer.resetGameState(); 
 74         
 75         console.LogInfo("Game created");
 76     },
 77     
 78     /** @param {null} */
 79     startGame: function()
 80     {
 81         if (!data.p1_client_id || !data.p2_client_id)
 82         {
 83             console.LogInfo("Cannot start game, players missing!");
 84             return;
 85         }
 86 
 87         // Give initial push
 88         // \todo randomize the direction
 89         var velVec = new float3(0, 0, 0);
 90         velVec.x = 10;
 91         velVec.z = 2;
 92         data.ball.rigidbody.SetLinearVelocity(velVec);
 93         
 94         data.speed = 16;
 95         data.gameTimeIter = 0.0;
 96         data.running = true;
 97     }, 
 98      
 99     /** @param {null} */
100     stopGame: function()
101     {
102         gameServer.resetGameState();
103         
104         data.running = false;
105         data.speed = 16;
106         data.winscore = 2;
107         data.gameTimeIter = 0.0;
108         data.p1_client_id = null;
109         data.p2_client_id = null;
110         data.p1_score = 0;
111         data.p2_score = 0;
112     },
113     
114     /** @param {number} scoreClientId */
115     finishRound: function(scoreClientId)
116     {
117         gameServer.resetGameState();
118         
119         if (data.p1_client_id == scoreClientId)
120             data.p1_score += 1;
121         else if (data.p2_client_id == scoreClientId)
122             data.p2_score += 1;
123         
124         var gameState = {};
125         gameState.key = "gamestate";
126         gameState.value= [{ "number" : 1, "score" : data.p1_score.toString() }, 
127                           { "number" : 2, "score" : data.p2_score.toString() }];
128         me.Exec(4, msgServerInfomation, JSON.stringify(gameState));
129         
130         if (data.p1_score == data.winscore)
131         {
132             gameServer.finishGame(scoreClientId);
133             return;
134         }
135         else if (data.p2_score == data.winscore)
136         {            
137             gameServer.finishGame(scoreClientId);
138             return;
139         }
140             
141         gameServer.startGame();
142     },
143     
144     /** @param {number} winnerClientId */
145     finishGame: function(winnerClientId)
146     {
147         gameServer.stopGame();
148         
149         var gameState = {};
150         gameState.key = "gamefinished";
151         gameState.value= {};
152         gameState.value.message = "Game finished - Winner was " + gameServer.getClientName(winnerClientId) + "\nGame is open for new players";
153         
154         me.Exec(4, msgServerInfomation, JSON.stringify(gameState));
155         
156         console.LogInfo("Game finished");
157     },
158     
159     /** @param {null} */
160     resetGameState: function()
161     {
162         // Reset ball
163         // \todo know the zero pos for ball in data
164         var ballt = data.ball.placeable.transform;
165         ballt.pos = zeroVec; 
166         ballt.rot = zeroVec;
167         data.ball.placeable.transform = ballt;
168         
169         // Reset speed 
170         data.ball.rigidbody.SetLinearVelocity(zeroVec);
171         
172         // Reset bat positions
173         var ta = data.p1_bat.placeable.transform;
174         var tb = data.p2_bat.placeable.transform;
175         var ball_z = data.ball.placeable.transform.pos.z;
176         ta.pos.y = ball_y;
177         tb.pos.y = ball_y;
178         data.p1_bat.placeable.transform = ta;
179         data.p2_bat.placeable.transform = tb;
180     },
181     
182     /** @param {number} frametime The time elapsed since previous frame. */
183     updateGame: function(frametime)
184     {
185         if (!data.running)
186             return;
187 
188         // Current game state
189         var t_p1 = data.p1_bat.placeable.transform;
190         var t_p2 = data.p2_bat.placeable.transform;
191         var t_b = data.ball.placeable.transform;
192         var p_p1 = t_p1.pos;
193         var p_p2 = t_p2.pos;
194         var p_b = t_b.pos;
195         var r_b = data.ball.rigidbody;
196 
197         // Update game speed
198         data.gameTimeIter += frametime;
199         if (data.gameTimeIter > 10.0)
200         {
201             data.speed *= 1.2;
202             console.LogInfo(">> Speeding up the game to " + data.speed.toString());
203             data.gameTimeIter = 0.0;
204         }
205         
206         // Update ball
207         var velVec = r_b.GetLinearVelocity();
208         velVec = velVec.normalize().mul(data.speed);
209         r_b.SetLinearVelocity(velVec);
210         
211         // Update ai bat(s)
212         if (data.p1_client_id == data.serverId)
213         {
214             t_p1.pos.y = p_b.y;
215             if (t_p1.pos.y > 10.0)
216                 t_p1.pos.y = 10.0
217             if (t_p1.pos.y < -10.0)
218                 t_p1.pos.y = -10;
219             data.p1_bat.placeable.transform = t_p1; 
220         }
221         if (data.p2_client_id == data.serverId)
222         {
223             t_p2.pos.y = p_b.y;
224             if (t_p2.pos.y > 10.0)
225                 t_p2.pos.y = 10.0
226             if (t_p2.pos.y < -10.0)
227                 t_p2.pos.y = -10;
228             data.p2_bat.placeable.transform = t_p2; 
229         }
230         
231         // Check if someone scored
232         if (p_b.x < p_p1.x)
233         {
234             gameServer.finishRound(data.p2_client_id);
235         }
236         if (p_b.x > p_p2.x)
237         {
238             gameServer.finishRound(data.p1_client_id);
239         }
240     },

242     // \todo add avatar/client name as param
243     /** @param {number} clientId */
244     onClientJoin: function(/**number*/ clientId)
245     {   
246         var response = {};
247         response.key = "clientjoin";
248         response.value = {};
249                 
250         if (data.p1_client_id == null)
251         {
252             data.p1_client_id = clientId;
253             response.value.result = true;
254             response.value.number = 1;
255         }
256         else if (data.p2_client_id == null)
257         {
258             data.p2_client_id = clientId;
259             response.value.result = true;
260             response.value.number = 2;
261         }
262         else
263         {
264             response.value.result = false;
265             response.value.number = 0;
266         }
267         
268         response.value.id = clientId;
269         response.value.name = gameServer.getClientName(clientId);       
270         me.Exec(4, msgServerInfomation, JSON.stringify(response));
271 
272         if (response.value.result)
273             console.LogInfo(">> Player #" + response.value.number + " is " + response.value.name);
274         
275         if (data.p1_client_id && data.p2_client_id)
276         {
277             var infoStartGame = {};
278             infoStartGame.key = "startgame";
279             infoStartGame.value= [{ "id" : data.p1_client_id, "name" : gameServer.getClientName(data.p1_client_id) }, 
280                                   { "id" : data.p2_client_id, "name" : gameServer.getClientName(data.p2_client_id) }];
281             me.Exec(4, msgServerInfomation, JSON.stringify(infoStartGame));
282             gameServer.startGame();
283         }

286     /** 
287      * @param {number} clientId The id of the client that moved.
288      * @param {string} string constant for the movement direction ("up" / "down")
289      */    
290     onClientMovement: function(/**number*/ clientId, /**string*/ direction)
291     {
292         if (!data.running)
293             return;
294             
295         var bat = null;
296         if (data.p1_client_id == clientId)
297             bat = data.p1_bat;
298         else if (data.p2_client_id == clientId)
299             bat = data.p2_bat;
300         else
301             return;
302             
303         var t = bat.placeable.transform;
304         if (direction == "up")
305             t.pos.y += 0.5;
306         else
307             t.pos.y -= 0.5;
308         if (t.pos.y > 10.0)
309             t.pos.y = 10.0
310         if (t.pos.y < -10.0)
311             t.pos.y = -10;
312         bat.placeable.transform = t;
313     }

346         widget.setParent(ui.MainWindow());

348         widget.button_join.clicked.connect(this.joinGamePushed);

350         
351         this.gameData.widget = widget;
352

354     },

357     /** @method */
358     resetClient: function()
359     {
360         data.widget.p1_name.text = "";
361         data.widget.p2_name.text = "";
362         data.widget.p1_score.text = "0";
363         data.widget.p2_score.text = "0";
364         data.widget.button_join.enabled = true;
365         data.widget.button_add_cpu.enabled = true;
366         this.gameData.running = false;
367     },
368     
369     /** @method */
370     joinGamePushed: function() 
371     {
372         if (data.running)
373             return;
374         
375         // Make a Local input mapper if there is none yet
376         if (!me.GetComponent("EC_InputMapper", "PongInput")) 
377         {        
378             var inputMapper = me.GetOrCreateComponent("EC_InputMapper", "PongInput", 2, false);
379             inputMapper.contextPriority = 101;
380             inputMapper.takeMouseEventsOverQt = false;
381             inputMapper.takeKeyboardEventsOverQt = false;
382             inputMapper.modifiersEnabled = false;
383             inputMapper.keyrepeatTrigger = true; 
384             inputMapper.executionType = 1;
385             inputMapper.RegisterMapping("R", "Move(up)", 1);
386             inputMapper.RegisterMapping("F", "Move(down)", 1);
387         }
388         me.Action("Move").Triggered.connect(gameClient.onMovement);
389         me.Exec(2, msgJoinGame, data.myId);
390     },
391     
392     /** @method */
393     addCpuPlayer: function()
394     {
395         if (data.running)
396             return;
397             
398         me.Exec(2, msgJoinGame, 0);
399     },
400

402      * @method
403      * @param {string} direction The movement direction string constant ("up"/"down")
404      */
405     onMovement: function(direction)
406     {
407         if (!data.running)
408             return;
409

413     /** 
414      * @method
415      * @param {string} jsonData
416      */
417     onServerData: function(jsonData)
418     {
419         var sd = JSON.parse(jsonData);
420         var myIdString = data.myId.toString();
421         
422         /////// JOIN
423         if (sd.key == "clientjoin")
424         {
425             if (sd.value.id == myIdString)
426             {
427                 if (sd.value.result)
428                 {
429                     data.widget.button_join.enabled = false;
430                 }
431                 else
432                     console.LogInfo(">> Could not join game");
433             }
434             if (sd.value.result)
435             {
436                 eval("data.widget.p" + sd.value.number + "_name.text = \"" + sd.value.name + "\"");
437                 if (data.widget.p1_name.text == "")
438                     data.widget.label_status.text = "Waiting for player 1";
439                 if (data.widget.p2_name.text == "")
440                     data.widget.label_status.text = "Waiting for player 2";
441             }
442         }
443         
444         ////// GAME STATE
445         if (sd.key == "gamestate")
446         {
447             for (var i in sd.value)
448             {
449                 var playerstate = sd.value[i];
450                 eval("data.widget.p" + playerstate.number + "_score.text = " + playerstate.score);
451             }
452         }
453         
454         if (sd.key == "gamefinished")
455         {
456             data.widget.label_status.text = sd.value.message;
457             gameClient.resetClient();
458         }
459         
460         ////// START GAME 
461         if (sd.key == "startgame")
462         {
463             data.widget.button_join.enabled = false;
464             data.widget.button_add_cpu.enabled = false;
465         
466             for (var i in sd.value)
467             {
468                 var player = sd.value[i];
469                 if (player.id == data.myId.toString())
470                 {
471                     console.LogInfo(">> Game started");                    
472                     data.running = true;
473                 }
474             }
475             
476             data.widget.label_status.text = "Game ongoing";
477         }
478     }
